94 template <typename T>
95 static BinNode<T>* merge(BinNode<T>* a, BinNode<T>* b)
96 {
97  if (!a) return b;
98  if (!b) return a;
99 
100  // 确保a >= b，因为合并后，a做为根节点，必须为最大值
101  if (a->data < b->data)
102  dsa::swap(a, b);
103  // 合并a的右子堆与b
104  a->right = merge(a->right, b); // 因为右侧链长度 d <= O(log(n))，故merge最多递归d次，即时间复杂度为O(log(n))
105  a->right->parent = a;
106  // 使a继续满足左倾性
107  if (!a->left || a->left->npl < a->right->npl)
108  dsa::swap(a->left, a->right);
109  // 更新npl，若a->right=nullptr，则a距a->right的距离为1
110  a->npl = a->right ? (a->right->npl + 1) : 1;
111 
112  return a;
113 }
