133 template <typename T>
134 void BTree<T>::solve_overflow(BTNodePtr<T> node)
135 {
136  // 根据分支数判断是否上溢
137  if (node->child.size() <= this->m_order)
138  return;
139 
140  // 判断是否为根节点
141  BTNodePtr<T> p = node->parent;
142  if (!p)
143  {
144  // 在在根节点上溢时，树高度会增加
145  this->m_root = p = new BTNode<T>(this->m_order); // BTNode默认构造函数，在没关键码时有一个指向nullptr子节点
146  p->child[0] = node;
147  node->parent = p;
148  }
149 
150  // 转移node中s之后关键码和分支节点，追加到新节点rc中
151  // 即转移示意图中的 *[5]* 部分
152  int s = this->m_order / 2; // 确定上移的节点位置
153  BTNodePtr<T> rc = new BTNode<T>(this->m_order); // BTNode默认构造函数，在没关键码时有一个指向nullptr子节点
154  rc->child.clear(); // 清除第一个默认添加的nullptr子节点
155  for (int k = s + 1; k < node->key.size(); k ++)
156  {
157  rc->key.push_back(node->key[k]);
158  rc->child.push_back(node->child[k]);
159  }
160  rc->child.push_back(node->child[node->child.size()-1]); // 分支节点数比关键码数多1
161  if (rc->child[0]) // 非叶节点的子节点均在存(不为nullptr)
162  {
163  for (int k = 0; k < rc->child.size(); k ++)
164  rc->child[k]->parent = rc; // 更新父节点，rc即在p->child中
165  }
166  node->key.remove(s+1, node->key.size());// 移除原节点node中的关键码和分支
167  node->child.remove(s+1, node->child.size());
168 
169  // 上移目标关键码
170  // 即上移示意图的中 [3]
171  int r = p->key.search(node->key[s]); // 确定移至父节点的位置
172  p->key.insert(r+1, node->key.remove(s));
173  p->child.insert(r+2, rc); // 连接节点，p->child[r+2]现在指向rc
174  rc->parent = p;
175 
176  solve_overflow(p);
177 }

215 template <typename T>
216 void BTree<T>::solve_underflow(BTNodePtr<T> node)
217 {
218  int order_floor = (this->m_order+1)/2;
219  if (node->child.size() >= order_floor)
220  return;
221 
222  BTNodePtr<T> p = node->parent;
223  if (!p)
224  {
225  if ((0 == node->key.size()) && node->child[0]) // 根节点不含关键码(即没有数据了)
226  {
227  // 在在根节点下溢时，树高度会下降
228  this->m_root = node->child[0];
229  this->m_root->parent = nullptr;
230  node->child[0] = nullptr;
231  delete node;
232  }
233  return;
234  }
235 
236  // 获取node在父节点中的位置，这里不能用search，因为node是指针，在vector中是无序的
237  // r在p->child的左右两侧，必有一侧存在
238  int r = p->child.find(node);
239 
240  // 尝试旋转
241  if (r > 0)
242  {
243  // 尝试旋转，向左侧子节点借
244  BTNodePtr<T> s = p->child[r-1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
245  if (s->child.size() > order_floor)
246  {
247  //int y = r-1;
248  node->key.push_front(p->key[r-1]);
249  node->child.push_front(s->child.remove(s->child.size()-1));
250  p->key[r-1] = s->key.remove(s->key.size()-1);
251  return;
252  }
253  }
254  if (r + 1 < p->child.size())
255  {
256  // 尝试旋转，向右侧子节点借
257  BTNodePtr<T> s = p->child[r+1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
258  if(s->child.size() > order_floor)
259  {
260  //int y = r;
261  node->key.push_back(p->key[r]);
262  node->child.push_back(s->child.remove(0));
263  p->key[r] = s->key.remove(0);
264  return;
265  }
266  }
267 
268  // 使用合并
269  if (r > 0)
270  {
271  // 左侧存在
272  BTNodePtr<T> s = p->child[r-1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
273  //int y = r-1;
274 
275  s->key.push_back(p->key.remove(r-1));
276  p->child.remove(r);
277 
278  for (int k = 0; k < node->key.size(); k++)
279  s->key.push_back(node->key[0]);
280 
281  for (int k = 0; k < node->child.size(); k++)
282  {
283  if (node->child[k])
284  node->child[k]->parent = s;
285  s->child.push_back(node->child[k]);
286  }
287  delete node;
288  }
289  else
290  {
291  // 右侧存在
292  BTNodePtr<T> s = p->child[r+1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
293  //int y = r;
294 
295  s->key.push_front(p->key.remove(r));
296  p->child.remove(r);
297 
298  for (int k = 0; k < node->key.size(); k++)
299  s->key.insert(k, node->key[k]);
300 
301  for (int k = 0; k < node->child.size(); k++)
302  {
303  if (node->child[k])
304  node->child[k]->parent = s;
305  s->child.insert(k, node->child[k]);
306  }
307  delete node;
308  }
309 
310  solve_underflow(p);
311 }

370 {
371  BTNodePtr<T> node = this->search(e);
372  if (node) return false;
373 
374  // 找到关键码e在m_hot插入的位置
375  int r = this->m_hot->key.search(e);
376  this->m_hot->key.insert(r+1, e);
377  this->m_hot->child.push_back(nullptr); // 因为叶节点所有子节点为nullptr，故插在任意位置均可
378  this->m_size++;

392 template <typename T>
393 bool BTree<T>::remove(const T& e)
394 {
395  BTNodePtr<T> node = this->search(e);
396  if (!node) return false;
397 
398  int r = node->key.search(e);
399  if (node->child[0])
400  {
401  // node不是叶节点，则e右侧必有子节点
402  BTNodePtr<T> u = node->child[r+1];
403  while(node->child[0]) u = u->child[0]; // 迭代至e的后继(后继必定是叶节点)
404  node->key[r] = u->key[0]; // 将u->key[0]与待删除e交换位置
405  //u->key[0] = e; // 赋不赋值无所谓，反正要删除
406  node = u; // node重新指向包含待删除目标e的叶节点u
407  r = 0; // r也指向e在node->key中的下标
408  }
409 
410  // 如果待删除目标e在叶节点中，直接删除
411  node->key.remove(r);
412  node->child.remove(node->child.size() - 1); // 因为叶节点所有子节点为nullptr，故随便删除一个即可
413  this->m_size--;
414 
415  this->solve_underflow(node);
416  return true;
417 }
