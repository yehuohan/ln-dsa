897 template <typename T>
898 void Vector<T>::shell_insertion(int w, int lo, int hi)
899 {
900 #if(1)
901  // 每w次循环，完成对所有列的同一行的插入排序
902  // 第一行元素： lo , lo+1 , ... lo+w-1
903  // 第二行元素： lo+w , lo+w+1 , ...
904  for (int k = lo + w; k < hi; k ++)
905  {
906  // 此部分程序参照insertion_sort
907  int n = k;
908  T tmp = this->m_array[n];
909  while (n >= lo+w && tmp < this->m_array[n-w])
910  {
911  this->m_array[n] = this->m_array[n-w];
912  n -= w;
913  }
914  this->m_array[n] = tmp;
915  }
916 
917 #else
918  // 依次对第 0,1,2...w 列进行排序(排完第0列，才能排第1列······)
919  int cow = ((hi-lo) + w - 1) / w; // 向上取整，即为矩阵的行数
920  for (int k = 0; k < w; k ++) // 对所有列进行插入排序
921  {
922  int sl = lo + k, sh = lo + k+cow*w; // (插入排序)列下标范围[sl,sh)
923  if (sh > hi) sh -= w; // 防止列下标超范围
924  for (int s = sl + w; s < sh; s += w)// 对列元素进行插入排序
925  {
926  // 此部分程序参照insertion_sort
927  int n = s;
928  T tmp = this->m_array[n];
929  while (n > sl && tmp < this->m_array[n-w])
930  {
931  this->m_array[n] = this->m_array[n-w];
932  n -= w;
933  }
934  this->m_array[n] = tmp;
935  }
936  }
937 #endif
938 }
