156 template <typename T>
157 static BinNode<T>* remove_at(
158  BinNode<T>*& node,
159  BinNode<T>*& hot)
160 {
161  BinNode<T>* w = node;
162  BinNode<T>* succ = nullptr;
163  // 只有右子树(或左右子树均没有，则返回nullptr，即被删除节点位置没有新节点)
164  if(!node->left) succ = node = node->right;
165  // 只有左子树
166  else if(!node->right) succ = node = node->left;
167  // 左右分支均存在
168  else
169  {
170  // 右子树存在，故node的直接后继w为node右子树最靠左（最小）的节点
171  w = w->successor();
172  // 交换直接后继w与目标节点的数据，交换完后，w成为待删除的目标节点
173  T tmp = w->data;
174  w->data = node->data;
175  node->data = tmp;
176  BinNode<T>* u = w->parent;
177  // 后继节点w只可能有右子树，不可能有左子树
178  if (u == node)
179  {
180  // 后继节点w即为node的右子节点
181  u->right = succ = w->right;
182  }
183  else
184  {
185  u->left = succ = w->right;
186  }
187  }
188 
189  hot = w->parent; // 待删除节点的父节点
190  // succ不为nullptr，则hot为succ的父节点
191  // succ为nullptr，则hot的子节点w被删除后，也为nullptr
192  if (succ) succ->parent = hot; // 设置接替节点的父节点
193  delete w; // 删除节点
194 
195  return succ;
196 }

289 template <typename T>
290 BinNode<T>* BinSearchTree<T>::rotate_at(BinNode<T>* v)
291 {
292  BinNode<T>* p = v->parent;
293  BinNode<T>* g = p->parent;
294  if (BN_IsLeftChild(*p))
295  {
296  if (BN_IsLeftChild(*v))
297  {
298  // zig 单旋
299  p->parent = g->parent; // 设置子树根节点的父节点
300  return this->connect34(v, p, g,
301  v->left, v->right, p->right, g->right);
302  }
303  else
304  {
305  // zig-zag 双旋
306  v->parent = g->parent; // 设置子树根节点的父节点
307  return this->connect34(p, v, g,
308  p->left, v->left, v->right, g->right);
309  }
310  }
311  else if (BN_IsRightChild(*p))
312  {
313  if (BN_IsLeftChild(*v))
314  {
315  // zag-zig 双旋
316  v->parent = g->parent; // 设置子树根节点的父节点
317  return this->connect34(g, v, p,
318  g->left, v->left, v->right, p->right);
319  }
320  else
321  {
322  // zag 单旋
323  p->parent = g->parent; // 设置子树根节点的父节点
324  return this->connect34(g, p, v,
325  g->left, p->left, v->left, v->right);
326  }
327  }
328 }
