380 template <typename T, typename VST>
381 static void traverse_LRD_iteration(BinNodePtr<T> node, VST& visit)
382 {
383  dsa::Stack<BinNodePtr<T> > s;
384  BinNodePtr<T> last_visited = nullptr;
385 
386  while(node)
387  {
388  s.push(node); node = node->left; // 将L入栈
389  }
390  while(!s.is_empty())
391  {
392  node = s.pop();
393  // LRD中的D节点能被访问的前提是：无右子树或右子树已被访问过
394  if (!node->right || node->right == last_visited )
395  {
396  visit(node->data);
397  last_visited = node;
398  }
399  else
400  {
401  // 将LRD中的D重新入栈，等待右子树被访问完
402  s.push(node);
403  node = node->right;
404  while(node)
405  {
406  s.push(node); node = node->left; // 将L入栈
407  }
408  }
409  }
410 }
