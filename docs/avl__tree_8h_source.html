110 {
111  BinNodePtr<T>& x = this->search(e);
112  if (x) return x; // 不插入重复节点
113 
114  x = new BinNode<T>(e, this->m_hot); // 新插入的节点
115  this->m_size ++;
116  BinNodePtr<T> node = x;
117 
118  // 从新插入节点的父节点开始，向上遍历所有父节点，检查是否满足Avl平衡
119  for(BinNodePtr<T> g = this->m_hot; g; g = g->parent)
120  {
121  if(!AVL_Balanced(*g)) // 所有失衡节点中的最低点
122  {
123  BinNodePtr<T>& sub_node = RefFromParent(*g);
124  // 获取g在父节点的孩子节点指针

126  // 将rotate_at返回子树的根节点添加到Avl树中
127  break;
128  }
129  else
130  this->update_height(g);
131  }
