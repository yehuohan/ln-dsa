622 int* build_gs(char* P)
623 {
624  int m = strlen(P);
625  int* gs = new int[m];
626  int* ss = new int[m];
627 
628 #if(0)
629  // build ss，蛮力计算 O(m^2)
630  for (int k = m-1; k >= 0; k--)
631  {
632  int j = k;
633  for (int n = m-1; n >= 0; n--)
634  {
635  if(P[n] != P[j])
636  {
637  // P的子串和后缀相等
638  ss[k] = k-j;
639  break;
640  }
641  if (--j < 0)
642  {
643  // P的前缀和后缀相等
644  ss[k] = k-j;
645  break;
646  }
647  }
648  }
649 #else
650  // 优化算法 O(m)
651  ss[m-1] = m; //对最后一个字符而言，与之匹配的最长后缀就是整个P串
652  for (int lo = m - 1, hi = m - 1, j = lo - 1; j >= 0; j --)
653  {
654  if (( lo < j) &&
655  (ss[m - (hi-j+1)] <= j - lo))
656  //情况一
657  ss[j] = ss[m - (hi-j+1)]; // 利用此前已计算出的ss[]
658  else
659  {
660  //情况二
661  hi = j;
662  lo = (lo < hi) ? lo : hi;
663  while ((0 <= lo) &&
664  (P[lo] == P[m - (hi-lo+1)]))
665  lo--; // 逐个对比处位于(lo, hi]前端的字符
666  ss[j] = hi - lo;
667  }
668  }
669 #endif
670 
671  // build gs
672  for (int k = 0; k < m; k ++)
673  gs[k] = m;
674  for (int i = 0, k = m-1; k >= 0; k--)
675  {
676  // (2.1)的情况
677  // 因为k是倒序遍历，所在先遇到的是较长的前缀，因而先计算ss的也是P中较前面的字符
678  if (ss[k] == k + 1)
679  {
680  while (i < m-k-1)
681  gs[i++] = m-(k+1);
682  }
683  }
684  for (int k = 0; k < m-1; k++)
685  gs[m-ss[k]-1] = m-k-1; // (2.2)的情况，不会影响(2.1)情况中计算的值
686 
687  delete[] ss;
688  return gs;
689 }
