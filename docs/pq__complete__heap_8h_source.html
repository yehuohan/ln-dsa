114 template <typename T>
115 int PqComplHeap<T>::proper_parent(int n, int i)
116 {
117 #define PQ_Bigger(i, j) (((this->m_array[i]) >= (this->m_array[j])) ? (i) : (j))
118  if (PQ_RightChildValid(n, i))
119  {
120  // i的左右孩子均存在
121  return PQ_Bigger(i, PQ_Bigger(PQ_LeftChild(i), PQ_RightChild(i)) );
122  }
123  else if(PQ_LeftChildValid(n, i))
124  {
125  // 只有i的左孩子存在
126  return PQ_Bigger(i, PQ_LeftChild(i));
127  }
128  else
129  return i;
130 #undef PQ_Bigger
131 }

140 template <typename T>
141 int PqComplHeap<T>::percolate_up(int i)
142 {
143  T ele = this->m_array[i];
144  while(PQ_ParentValid(i))
145  {
146  int j = PQ_Parent(i);
147  if (ele < this->m_array[j])
148  break;
149  // 新插入元素比父元素大，则上滤元素
150  this->m_array[i] = this->m_array[j];
151  i = j;
152  }
153  // 将新插入元素，放到最终的节点位置
154  this->m_array[i] = ele;
155  return i;
156 }

168 template <typename T>
169 int PqComplHeap<T>::percolate_down(int n, int i)
170 {
171  int j;
172  while(i!= (j = this->proper_parent(n, i)))
173  {
174  // i的子节点元比i大，则下滤
175  dsa::swap(this->m_array[i], this->m_array[j]);
176  i = j;
177  }
178  return i;
179 }

235 template <typename T>
236 void PqComplHeap<T>::heapify(int n)
237 {
238 #if(0)
239  // 按自上而下的顺序上滤
240  // 效率太低：对所有节点的深度求和
241  // 越往堆底，深度越大（上滤的距离越大），节点数越多
242  // Sum[depth(i)] = O(nlog(n))
243  for (int k = 1; k < n; k ++)
244  {
245  this->percolate_up(k);
246  }
247 
248 #else

250  // 按自下而上的顺序下滤
251  // PQ_Parent(n-1) ：最后一个内部节点，也即末节点的父亲
252  // 效率较高： 对所有节点的高度求和
253  // 越往堆底，高度越小（下滤的距离越小），节点数越多
254  // Sum[height(i)] = O(n)
255  for(int k = PQ_Parent(n-1); k >= 0; k--)
256  {
257  this->percolate_down(n, k);
