86 template <typename T>
87 BinNode<T>*& SplayTree<T>::search(const T& e)
88 {
89  BinNode<T>* p = search_in(this->m_root, e, this->m_hot = nullptr);
90  // 无论是否找到节点，均会进行伸展操作，即没有找到目标节点，也将靠近目标的节点移到树根
91  this->m_root = this->splay(p ? p : this->m_hot);
92  // 若m_root为nullptr（即没有查找到目标e，且m_hot也为nullptr）
93  // 则一定是空树的情况（非空树，即使没有查找到目录，m_hot也不会为nullptr）
94  return this->m_root;
95 }

104 template <typename T>
105 BinNode<T>* SplayTree<T>::insert(const T& e)
106 {
107  // 如果是空树，直接插入节点即可
108  if (!this->m_root)
109  {
110  this->m_size++;
111  this->m_root = new BinNode<T>(e, nullptr);
112  return this->m_root;
113  }
114 
115  // 因为不是空树,返回的x不可以为nullptr
116  // search 返回经过伸展后的树根节点
117  if (this->search(e)->data == e) return this->m_root;
118 
119  BinNode<T>* r = this->m_root;
120  this->m_root = new BinNode<T>(e, nullptr);
121  this->m_size++;
122 
123  // 确定新插入节点的位置
124  if (e < r->data)
125  {
126  attach_left(this->m_root, r->left);
127  attach_right(this->m_root, r);
128  r->parent = this->m_root;
129  r->left = nullptr;
130 
131  }
132  else if (r->data < e)
133  {
134  attach_right(this->m_root, r->right);
135  attach_left(this->m_root, r);
136  r->parent = this->m_root;
137  r->right = nullptr;
138  }
139  this->update_height_above(r);
140 
141  return this->m_root;
142 }

151 template <typename T>
152 bool SplayTree<T>::remove(const T& e)
153 {
154  // 空树直接返回
155  if (!this->m_root) return false;
156  // 没有找到目标
157  if (this->search(e)->data != e) return false;
158 
159  // 删除目标节点（经过伸展，已经移到了根节点）
160  BinNode<T>* s = this->m_root;
161  if (!BN_HasLeftChild(*this->m_root))
162  {
163  this->m_root = this->m_root->right;
164  if (this->m_root) this->m_root->parent = nullptr;
165  }
166  else if (!BN_HasRightChild(*this->m_root))
167  {
168  this->m_root = this->m_root->left;
169  if (this->m_root) this->m_root = nullptr;
170  }
171  else
172  {
173  BinNode<T>* lt = this->m_root->left;
174  // 暂时分离左子树
175  this->m_root->left = nullptr;
176  lt->parent = nullptr;
177  // 只保留右子树
178  this->m_root = this->m_root->right;
179  this->m_root->parent = nullptr;
180  // 以原树根为目标，在原右子树做一次（必定失败的）查找，
181  // 原右子树中的最大值经过伸展后，会移到原右子树的根节点
182  this->search(s->data);
183  // 再将原左子树连接到根节点
184  this->m_root->left = lt;
185  lt->parent = this->m_root;
186  }
187 
188  delete s;
189  this->m_size--;
190  if (this->m_root) this->update_height(this->m_root);
191 
192  return true;
193 }

231 template <typename T>
232 BinNode<T>* SplayTree<T>::splay(BinNode<T>* v)
233 {
234  if (!v) return nullptr;
235  BinNode<T>* p;
236  BinNode<T>* g;
237 
238  // 双层伸展，伸展完成后，v将成为子树的根节点
239  while((p = v->parent) && (g = p->parent))
240  {
241  BinNode<T>* gg = g->parent;
242  if (BN_IsLeftChild(*v))
243  {
244  if (BN_IsLeftChild(*p))
245  {
246  // zig-zig
247  attach_left(p, v->right);
248  attach_left(g, p->right);
249  attach_right(v, p);
250  attach_right(p, g);
251  }
252  else
253  {
254  // zig-zag
255  this->connect34(g, v, p,
256  g->left, v->left, v->right, p->right);
257  }
258  }
259  else
260  {
261  if (BN_IsRightChild(*p))
262  {
263  // zag-zag
264  attach_right(p, v->left);
265  attach_right(g, p->left);
266  attach_left(v, p);
267  attach_left(p, g);
268  }
269  else
270  {
271  // zag-zig
272  this->connect34(g, v, g,
273  p->left, v->left, v->right, g->right);
274  }
275 
276  }
277 
278  // 将新的子树接入到原树的位置
279  if (!gg)
280  v->parent = nullptr; // 成为树根节点
281  else
282  (g == gg->left) ? attach_left(gg, v) : attach_right(gg, v);
283  // 插入到原树作为左子树或右子树
284 
285  this->update_height(g);
286  this->update_height(p);
287  this->update_height(v);
288  }
289 
290  // 单层伸展，最多一次旋转
291  if ((p = v->parent))
292  {
293  if (BN_IsLeftChild(*v))
294  {
295  attach_left(p, v->right);
296  attach_right(v, p);
297  }
298  else
299  {
300  attach_right(p, v->left);
301  attach_left(v, p);
302  }
303  }
304 
305  // v成为树的根节点
306  v->parent = nullptr;
307  return v;
308 }
