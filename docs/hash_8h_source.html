239 template <typename K, typename V, typename HF, typename CMP>
240 bool HashTable<K,V,HF,CMP>::put(K key, V val)
241 {
242  // 先检测key是否存在，若已存在key，则放弃插入key-val
243  if (this->m_ht[this->probe_hit(key)])
244  return false;
245  // 试探出空单元，用于插入key-val
246  uint index = this->probe_free(key);
247  this->m_ht[index] = new Entry<K,V,CMP>(key, val);
248  this->m_size ++;
249  // 装填因子 >50% 时，重散列，保证恒有一定的空单元
250  if (this->m_size * 2 > this->m_cap)
251  this->rehash();
252  return true;
253 }

280 bool HashTable<K,V,HF,CMP>::remove(K key)
281 {

283  int index = this->probe_hit(key);

308 template <typename K, typename V, typename HF, typename CMP>
309 void HashTable<K,V,HF,CMP>::rehash()
310 {
311  int old_cap = this->m_cap;
312  dsa::Entry<K,V,CMP>** old_ht = this->m_ht;
313  // 重新初始化散列单元
314  delete this->lazy_rm;
315  this->init(2*old_cap);
316  // 转移散列单元
317  for (int k = 0; k < old_cap; k ++)
318  if (old_ht[k]) this->put(old_ht[k]->key, old_ht[k]->value);
319  // 释放原有散列单元
320  for (int k = 0; k < old_cap; k ++)
321  if (old_ht[k]) delete old_ht[k];
322  delete[] old_ht;
323 }

347 template <typename K, typename V, typename HF, typename CMP>
348 int HashTable<K,V,HF,CMP>::probe_line_hit(const K& key)
349 {
350  int r = this->hash_func(key) % this->m_cap;
351  while((this->m_ht[r] && *(this->m_ht[r]) != key) // 试探：跳过冲突的单元（优先跳过不为nullptr的单元）
352  ||(!this->m_ht[r] && Is_Lazy_Removed(r))) // 试探：跳过查找链上带懒惰删除标记的单元
353  {
354  r = (r + 1) % this->m_cap;
355  }
356  return r;
357 }

384 template <typename K, typename V, typename HF, typename CMP>
385 int HashTable<K,V,HF,CMP>::probe_quad_hit(const K& key)
386 {
387  int i = this->hash_func(key) % this->m_cap;
388  int r = i, s = 1;
389  while((this->m_ht[r] && *(this->m_ht[r]) != key) // 试探：跳过冲突的单元（优先跳过不为nullptr的单元）
390  ||(!this->m_ht[r] && Is_Lazy_Removed(r))) // 试探：跳过查找链上带懒惰删除标记的单元
391  {
392  if (s > 0)
393  r = (i + s*s) % this->m_cap;
394  else
395  r = (i - s*s) % this->m_cap;
396  s = (s > 0) ? -s : -(--s);
397  }
398  return r;
399 }
