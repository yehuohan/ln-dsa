135 template <typename T>
136 void RedBlackTree<T>::solve_double_red(BinNode<T>* x)
137 {
138  if (BN_IsRoot(*x))
139  {
140  x->color = RBColor::Black;
141  this->m_root->height++; // 变成黑节点，高度加1
142  return;
143  }
144  BinNode<T>* p = x->parent;
145  if (BN_IsBlack(p)) // p为黑节点，则没有双红问题
146  return;
147  BinNode<T>* g = p->parent; // p为红节点，必有黑父节点
148  BinNode<T>* u = BN_IsLeftChild(*p) ? g->right : g->left;
149 
150  if (BN_IsBlack(u))
151  {
152  // RR-1
153  g->color = RBColor::Red;
154  // x,p同侧，即同为左孩子或同为右孩子
155  if (BN_IsLeftChild(*x) == BN_IsLeftChild(*p))
156  p->color = RBColor::Black;
157  else
158  x->color = RBColor::Black;
159 
160  BinNode<T>*& n = RefFromParent(*g); // 获取g在父节点的孩子节点指针
161  n = this->rotate_at(x); // 34重构中会更新高度，故重染色需要在rotate_at之前进行
162  //n->color = RBColor::Black;
163  //n->left->color = RBColor::Red;
164  //n->right->color = RBColor::Red;
165  }
166  else
167  {
168  // RR-2 重新染色
169  // u原来为红节点，故一定不为nullptr
170  p->color = RBColor::Black;
171  p->height ++; // 由红转黑，高度加1
172  u->color = RBColor::Black;
173  u->height ++; // 由红转黑，高度加1
174  if (!BN_IsRoot(*g))
175  g->color = RBColor::Red; // 若g不是根节点，需要转红，继续检测双红问题
176  solve_double_red(g); // 若g是根节点，在递归调用中，高度会得到更新
177  }
178 }

247 template <typename T>
248 void RedBlackTree<T>::solve_double_black(BinNode<T>* r)
249 {
250  // 如果r为nullptr，则r的父节点为m_hot
251  BinNode<T>* p = r ? r->parent : this->m_hot;
252  // r为根节点
253  if (!p)
254  return;
255  // r的兄弟节点
256  BinNode<T>* s = (r == p->left) ? p->right : p->left;
257 
258  if (BN_IsBlack(s)) // s为黑
259  {
260  BinNode<T>* t = nullptr;
261  if (BN_IsRed(s->left)) t = s->left;
262  else if (BN_IsRed(s->right)) t = s->right;
263  if (t) // s至少有一个红子节点
264  {
265  // BB-1
266  RBColor clr = p->color;
267  BinNode<T>*& n = RefFromParent(*p);
268  // 因为34重构前，未进行颜色设置，故rotate_at中的高度更新无效
269  n = this->rotate_at(t);
270  // 重染色后，需要重新计算黑高度
271  if (BN_HasLeftChild(*n))
272  {
273  n->left->color = RBColor::Black;
274  this->update_height(n->left);
275  }
276  if (BN_HasRightChild(*n))
277  {
278  n->right->color = RBColor::Black;
279  this->update_height(n->right);
280  }
281  n->color = clr;
282  this->update_height(n);
283  }
284  else // s两个子节点均为黑
285  {
286  s->color = RBColor::Red; // s转红，高度减1
287  s->height--;
288  if (BN_IsRed(p)) // p为红
289  {
290  // BB-2R
291  p->color = RBColor::Black; //p转黑，但黑高度不变
292  }
293  else // p为黑
294  {
295  // BB-2B
296  p->height--; //p保持黑，但黑高度下降
297  solve_double_black(p);
298  }
299  }
300  }
301  else // s为红，两个子节点则一定为黑
302  {
303  // BB-3
304  p->color = RBColor::Red;
305  s->color = RBColor::Black;
306  BinNode<T>* t = (BN_IsLeftChild(*s)) ? s->left : s->right;
307  // 取t与s取同侧
308  this->m_hot = p;
309  BinNode<T>*& n = RefFromParent(*p);
310  n = this->rotate_at(t);
311  // 递归修正r处的双黑问题
312  solve_double_black(r);
313  }
314 }

324 BinNode<T>* RedBlackTree<T>::insert(const T& e)
325 {
326  BinNode<T>*& x = this->search(e); // 使用BinSearchTree::search()
327  if (x)

366 template <typename T>
367 bool RedBlackTree<T>::remove(const T& e)
368 {
369  BinNode<T>*& x = this->search(e);
370  if (!x)
371  return false;
372 
373  // r为接替x所在位置的节点，r可以为nullptr，m_hot为r的父节点
374  BinNode<T>* r = remove_at(x, this->m_hot);
375  // 没有节点了，则直接返回
376  if(!(--this->m_size))
377  return true;
378 
379  // 删除的是根节点，则r成为新的根节点
380  if(!this->m_hot)
381  {
382  this->m_root->color = RBColor::Black;
383  this->update_height(this->m_root);
384  return true;
385  }
386 
387  //若所有祖先的黑深度依然平衡，则无需调整
388  if (BN_BlackHeightUpdated(*this->m_hot))
389  return true;
390  // r为红，只需要染黑即可
391  if (BN_IsRed(r))
392  {
393  r->color = RBColor::Black;
394  r->height ++;
395  return true;
396  }
397  // 双黑问题
398  solve_double_black(r);
399  return true;
400 }
